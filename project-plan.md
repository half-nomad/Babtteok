```javascript
// 기본 구조: 순수 바닐라 JS + CSS
```

```css
/* backdrop-filter 지원 브라우저 우선 */
/* fallback: 반투명 배경색으로 대체 */
/* 네온 글로우: box-shadow 대신 filter 활용으로 성능 개선 */
```

```css
/* Desktop: 1920px+ (16:9 최적화) */
```
